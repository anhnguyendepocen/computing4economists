\begin{column}{.48\textwidth}
\color{blue}\tiny
\begin{verbatim}
mkdir <repo> && cd <repo>
git init
git remote add -f <name> <url>
git config core.sparsecheckout \
 true
echo some/dir/ >> .git/info/sparse-checkout
echo another/sub/tree >> \
 .git/info/sparse-checkout
git pull <remote> <branch>
\end{verbatim}
